99 //
100 // Bind target socket to initiator socket
101 // - Binds the port of the initiator socket to the export of the target
102 // socket
103 // - Binds the port of the target socket to the export of the initiator
104 // socket
105 //
106 virtual void bind(base_initiator_socket_type& s)
107 {
108 // initiator.port -> target.export
109 (s.get_base_port())(get_base_interface());
110 // target.port -> initiator.export
111 get_base_port()(s.get_base_interface());
112 }
113
114 void operator() (base_initiator_socket_type& s)
115 {
116 bind(s);
117 }
118
119 //
120 // Bind target socket to target socket (hierarchical bind)
121 // - Binds both the export and the port
122 //
123 virtual void bind(base_type& s)
124 {
125 // export
126 (get_base_export())(s.get_base_export());
127 // port
128 (s.get_base_port())(get_base_port());
129 }
130
131 void operator() (base_type& s)
132 {
133 bind(s);
134 }
135
136 //
137 // Bind interface to socket
138 // - Binds the interface to the export
139 //
140 virtual void bind(fw_interface_type& ifs)
141 {
142 export_type* exp = &get_base_export();
143 if( this == exp ) {
144 export_type::bind( ifs ); // non-virtual function call
145 } else {
146 exp->bind( ifs );
147 }
148 }
149
150 void operator() (fw_interface_type& s)
151 {
152 bind(s);
153 }
