33/*
34This class implements a trivial multi initiator socket.
35The triviality refers to the fact that the socket does not
36do blocking to non-blocking or non-blocking to blocking conversions.
37
38It allows to connect multiple targets to this socket.
39The user has to register callbacks for the bw interface methods
40he likes to use. The callbacks are basically equal to the bw interface
41methods but carry an additional integer that indicates to which
42index of this socket the calling target is connected.
43*/
44template <typename MODULE,
45 unsigned int BUSWIDTH = 32,
46 typename TYPES = tlm::tlm_base_protocol_types,
47 unsigned int N=0, sc_core::sc_port_policy POL = sc_core::SC_ONE_OR_MORE_BOUND>
48class multi_passthrough_initiator_socket
49 : public multi_init_base< BUSWIDTH, TYPES, N, POL>
50{
51public:
52
53 //typedefs
54 // tlm 2.0 types for nb_transport
55 typedef typename TYPES::tlm_payload_type transaction_type;
56 typedef typename TYPES::tlm_phase_type phase_type;
57 typedef tlm::tlm_sync_enum sync_enum_type;
58
59 // typedefs to keep the fn ptr notations short
60 typedef sync_enum_type (MODULE::*nb_cb)(int,
61 transaction_type&,
62 phase_type&,
63 sc_core::sc_time&);
64 typedef void (MODULE::*dmi_cb)(int, sc_dt::uint64, sc_dt::uint64);
65
66 typedef multi_init_base<BUSWIDTH, TYPES, N, POL> base_type;
67
68 typedef typename base_type::base_target_socket_type base_target_socket_type;
69
70 static const char* default_name()
71 { return sc_core::sc_gen_unique_name("multi_passthrough_initiator_socket"); }
72
73 //CTOR
74 explicit multi_passthrough_initiator_socket(const char* name = default_name())
75 : base_type(name)
76 , m_hierarch_bind(0)
77 , m_beoe_disabled(false)
78 , m_dummy(this,42)
79 {
80 }
81
82 ~multi_passthrough_initiator_socket(){
83 //clean up everything allocated by 'new'
84 for (unsigned int i=0; i<m_binders.size(); i++) delete m_binders[i];
85 }
86
87 //register callback for nb transport of bw interface
88 void register_nb_transport_bw(MODULE* mod,
89 sync_enum_type (MODULE::*cb)(int,
90 transaction_type&,
91 phase_type&,
92 sc_core::sc_time&))
93 {
94 //warn if there already is a callback
95 if (m_nb_f.is_valid()){
96 display_warning("NBTransport_bw callback already registered.");
97 return;
98 }
99
100 //set the functor
101 m_nb_f.set_function(mod, cb);
102 }
103
104 //register callback for dmi function of bw interface
105 void register_invalidate_direct_mem_ptr(MODULE* mod,
106 void (MODULE::*cb)(int, sc_dt::uint64, sc_dt::uint64))
107 {
108 //warn if there already is a callback
109 if (m_dmi_f.is_valid()){
110 display_warning("InvalidateDMI callback already registered.");
111 return;
112 }
113
114 //set the functor
115 m_dmi_f.set_function(mod, cb);
116 }
117
118 //Override virtual functions of the tlm_initiator_socket:
119 // this function is called whenever an sc_port (as part of a target socket)
120 // wants to bind to the export of the underlying tlm_initiator_socket
121 //At this time a callback binder is created an returned to the sc_port
122 // of the target socket, so that it binds to the callback binder
123 virtual tlm::tlm_bw_transport_if<TYPES>& get_base_interface()
124 {
125 m_binders.push_back(new callback_binder_bw<TYPES>(this, m_binders.size()));
126 return *m_binders[m_binders.size()-1];
127 }
128
129 // const overload not allowed for multi-sockets
130 virtual const tlm::tlm_bw_transport_if<TYPES>& get_base_interface() const
131 {
132 display_error("'get_base_interface()' const not allowed for multi-sockets.");
133 return base_type::get_base_interface();
134 }
135
136 //Override virtual functions of the tlm_initiator_socket:
137 // this function is called whenever an sc_export (as part of a initiator socket)
138 // wants to bind to the export of the underlying tlm_initiator_socket
139 // i.e. a hierarchical bind takes place
140 virtual sc_core::sc_export<tlm::tlm_bw_transport_if<TYPES> >& get_base_export()
141 {
142 if (!m_beoe_disabled) //we are not bound hierarchically
143 base_type::m_export.bind(m_dummy); //so we bind the dummy to avoid a SystemC error
144 return base_type::get_base_export(); //and then return our own export so that the hierarchical binding is set up properly
145 }
146
147 virtual const sc_core::sc_export<tlm::tlm_bw_transport_if<TYPES> >& get_base_export() const
148 {
149 return base_type::get_base_export();
150 }
151
152 //bind against a target socket
153 virtual void bind(base_target_socket_type& s)
154 {
155 //error if this socket is already bound hierarchically
156 if (m_hierarch_bind) {
157 display_error("Already hierarchically bound.");
158 return;
159 }
160
161 base_type::bind(s); //satisfy systemC, leads to a call to get_base_interface()
162
163 //try to cast the target socket into a fw interface
164 sc_core::sc_export<tlm::tlm_fw_transport_if<TYPES> >* p_ex_s=dynamic_cast<sc_core::sc_export<tlm::tlm_fw_transport_if<TYPES> >*>(&s);
165 if (!p_ex_s) {
166 display_error("Multi socket not bound to tlm_socket.");
167 return;
168 }
169
170 //try a cast into a multi sockets
171 multi_to_multi_bind_base<TYPES>* test=dynamic_cast<multi_to_multi_bind_base<TYPES>*> (p_ex_s);
172 if (test) //did we just do a multi-multi bind??
173 //if that is the case the multi target socket must have just created a callback binder
174 // which we want to get from it.
175 //Moreover, we also just created one, which we will pass to it.
176 m_sockets.push_back(test->get_last_binder(m_binders[m_binders.size()-1]));
177 else{ // if not just bind normally
178 sc_core::sc_export<tlm::tlm_fw_transport_if<TYPES> >& ex_s=*p_ex_s;
179 m_sockets.push_back(&((tlm::tlm_fw_transport_if<TYPES>&)ex_s)); //store the interface we are bound against
180 }
181 }
182
183 //operator notation for direct bind
184 void operator() (base_target_socket_type& s)
185 {
186 bind(s);
187 }
188
189 //SystemC standard callback before end of elaboration
190 void before_end_of_elaboration(){
191 //if our export hasn't been bound yet (due to a hierarch binding)
192 // we bind it now to avoid a SystemC error.
193 //We must do that, because it is legal not to register a callback on this socket
194 // as the user might only use b_transport
195 if (!base_type::m_export.get_interface()){
196 base_type::m_export.bind(m_dummy);
197 }
198
199 //'break' here if the socket was told not to do callback binding
200 if (m_beoe_disabled) return;
201
202 //get the callback binders of the top of the hierachical bind chain
203 // NOTE: this could be the same socket if there is no hierachical bind
204 std::vector<callback_binder_bw<TYPES>* >& binders=get_hierarch_bind()->get_binders();
205
206 //get the interfaces bound to the top of the hierachical bind chain
207 // NOTE: this could be the same socket if there is no hierachical bind
208 m_used_sockets=get_hierarch_bind()->get_sockets();
209
210 //register the callbacks of this socket with the callback binders
211 // we just got from the top of the hierachical bind chain
212 for (unsigned int i=0; i<binders.size(); i++) {
213 binders[i]->set_callbacks(m_nb_f, m_dmi_f);
214 }
215 }
216
217 //
218 // Bind multi initiator socket to multi initiator socket (hierarchical bind)
219 //
220 virtual void bind(base_type& s)
221 {
222 if (m_binders.size()) {
223 //a multi socket is either bound hierarchically or directly
224 display_error("Socket already directly bound.");
225 return;
226 }
227 if (m_hierarch_bind){
228 display_warning("Socket already bound hierarchically. Bind attempt ignored.");
229 return;
230 }
231
232 //remember to which socket we are hierarchically bound and disable it,
233 // so that it won't try to register callbacks itself
234 s.disable_cb_bind();
235 m_hierarch_bind=&s;
236 base_type::bind(s); //satisfy SystemC
237 }
238
239 //operator notation for hierarchical bind
240 void operator() (base_type& s)
241 {
242 bind(s);
243 }
244
245 //get access to sub port
246 tlm::tlm_fw_transport_if<TYPES>* operator[](int i){return m_used_sockets[i];}
247
248 //get the number of bound targets
249 // NOTE: this is only valid at end of elaboration!
250 unsigned int size() {return get_hierarch_bind()->get_sockets().size();}
251
252protected:
253 using base_type::display_warning;
254 using base_type::display_error;
255
256 //implementation of base class interface
257 base_type* get_hierarch_bind(){if (m_hierarch_bind) return m_hierarch_bind->get_hierarch_bind(); else return this;}
258 void disable_cb_bind(){ m_beoe_disabled=true;}
259 std::vector<callback_binder_bw<TYPES>* >& get_binders(){return m_binders;}
260 std::vector<tlm::tlm_fw_transport_if<TYPES>*>& get_sockets(){return m_sockets;}
261 //vector of connected sockets
262 std::vector<tlm::tlm_fw_transport_if<TYPES>*> m_sockets;
263 std::vector<tlm::tlm_fw_transport_if<TYPES>*> m_used_sockets;
264 //vector of binders that convert untagged interface into tagged interface
265 std::vector<callback_binder_bw<TYPES>*> m_binders;
266
267 base_type* m_hierarch_bind; //pointer to hierarchical bound multi port
268 bool m_beoe_disabled; // bool that remembers whether this socket shall bind callbacks or not
269 callback_binder_bw<TYPES> m_dummy; //a callback binder that is bound to the underlying export
270 // in case there was no real bind
271
272 //callbacks as functors
273 // (allows to pass the callback to another socket that does not know the type of the module that owns
274 // the callbacks)
275 typename callback_binder_bw<TYPES>::nb_func_type m_nb_f;
276 typename callback_binder_bw<TYPES>::dmi_func_type m_dmi_f;
277};
