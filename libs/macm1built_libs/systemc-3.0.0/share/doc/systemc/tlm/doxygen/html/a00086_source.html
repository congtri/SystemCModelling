34tlm_fifo<T>::get( tlm_tag<T> * )

88tlm_fifo<T>::put( const T& val_ )
89{
90 while( is_full() ) {
91 wait( m_data_read_event );
92 }
93
94 if( buffer.is_full() ) {
95
96 buffer.resize( buffer.size() * 2 );
97
98 }
99
100 m_num_written ++;
101 buffer.write( val_ );
102
103 request_update();
104}
