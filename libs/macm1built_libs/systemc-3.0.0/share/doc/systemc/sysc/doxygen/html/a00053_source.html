235inline
236sc_bv_base::value_type
237sc_bv_base::get_bit( int i ) const
238{
239 int wi = i / SC_DIGIT_SIZE;
240 int bi = i % SC_DIGIT_SIZE;
241 return value_type( (m_data[wi] >> bi) & SC_DIGIT_ONE );
242}
243
244inline
245void
246sc_bv_base::set_bit( int i, value_type value )
247{
248 int wi = i / SC_DIGIT_SIZE;
249 int bi = i % SC_DIGIT_SIZE;
250 sc_digit mask = SC_DIGIT_ONE << bi;
251 m_data[wi] |= mask; // set bit to 1
252 m_data[wi] &= value << bi | ~mask;
253}
254
255
256inline
257void
258sc_bv_base::set_cword( int /*i*/, sc_digit w )
259{
260 if( w ) {
261 SC_REPORT_WARNING( sc_core::SC_ID_SC_BV_CANNOT_CONTAIN_X_AND_Z_, 0 );
262 }
263}
264
265
266inline
267void
268sc_bv_base::clean_tail()
269{
270 int wi = m_size - 1;
271 int bi = m_len % SC_DIGIT_SIZE;
272 if ( bi != 0 ) m_data[wi] &= ~SC_DIGIT_ZERO >> (SC_DIGIT_SIZE - bi);
273}
