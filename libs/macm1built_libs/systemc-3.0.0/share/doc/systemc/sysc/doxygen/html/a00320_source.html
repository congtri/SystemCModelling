137
138template <int W>
139void
140sc_in_rv<W>::end_of_elaboration()
141{
142 base_type::end_of_elaboration();
143 // check if bound channel is a resolved signal

145 this->report_error( SC_ID_RESOLVED_PORT_NOT_BOUND_, 0 );
146 }
147}
