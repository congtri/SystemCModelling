35template<int W>
36template<int W1>
37inline
38const sc_uint<W>& sc_uint<W>::operator = ( const sc_bigint<W1>& source )
39{
40 if ( W1 < 33 ) {
41 m_val = (int)*source.digit;
42 }
43 else {
44 m_val = ( (int64)source.digit[1] << BITS_PER_DIGIT ) | *source.digit;
45 }
46 return *this;
47}
