397inline
398double
399to_double() const
400{
401 return sc_signed_proxy().to_double();

404inline
405int
406to_int() const
407{
408 int result;
409
410 result = (int)digit[0];
411 return result;
412}
413
414inline
415unsigned int
416to_uint() const
417{
418 unsigned int result;
419
420 result = (unsigned int)digit[0];
421 return result;
422}
423
424inline
425int64
426to_int64() const
427{
428 int64 result;
429
430 if ( W < 33 ) {
431 result = to_int();
432 }
433 else {
434 result = ( (uint64)digit[1] << BITS_PER_DIGIT ) | digit[0];
435 }
436 return result;
437}
438
439inline
440uint64
441to_uint64() const
442{
443 uint64 result;
444
445 if ( W < 33 ) {
446 result = to_int();
447 }
448 else {
449 result = to_int64(); // @@@@#### ( (uint64)digit[1] << BITS_PER_DIGIT ) | digit[0];
450 }
451 return result;
452}

454inline
455long
456to_long() const
457{
458 long result = ( sizeof(long) < 5 ) ? to_int() : to_int64();
459 return result;
460}

465to_ulong() const
